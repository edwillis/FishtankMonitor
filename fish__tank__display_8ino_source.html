73 // Measure temperature from the thermistor, taking multiple readings to
74 // arrive at a more consistent value
75 float getTemp()
76 {
77  float reading = 0.0;
78  int i = 0;
79  for (i = 0; i<NUM_TEMP_SAMPLES; i++)
80  {
81  float next = analogRead(THERMISTORPIN);
82  reading += next;
83  delay(WAIT_BETWEEN_TEMP_SAMPLES);
84  }
85  reading /= (float) NUM_TEMP_SAMPLES;
86  // convert the value to resistance
87  reading = (float) 1023 / reading - 1;
88  reading = SERIESRESISTOR / reading;
89 
90  float steinhart;
91  steinhart = reading / THERMISTORNOMINAL; // (R/Ro)
92  steinhart = log(steinhart); // ln(R/Ro)
93  steinhart /= BCOEFFICIENT; // 1/B * ln(R/Ro)
94  steinhart += 1.0 / (TEMPERATURENOMINAL + 273.15); // + (1/To)
95  steinhart = 1.0 / steinhart; // Invert
96  steinhart -= 273.15; // convert to C
97  return steinhart;
98 }

100 float getPh()
101 {
102  float ph=0.0;
103  float voltage=0.0;
104  float measurement=0.0;
105  float next = 0.0;
106  int i = 0;
107  for (i=0; i<NUM_PH_SAMPLES; i++)
108  {
109  next = analogRead(PHPIN);
110  measurement += next;
111  delay(WAIT_BETWEEN_PH_SAMPLES);
112  }
113  measurement /= (float) NUM_PH_SAMPLES;
114  voltage = measurement * 5.0 / 1024.0;
115  ph = 3.5 * voltage + OFFSET;
