134 float getTemp()
135 {
136  float reading = 0.0;
137  int i = 0;
138  for (i = 0; i<NUM_TEMP_SAMPLES; i++)
139  {
140  float next = analogRead(THERMISTORPIN);
141  reading += next;
142  delay(WAIT_BETWEEN_TEMP_SAMPLES);
143  }
144  reading /= (float) NUM_TEMP_SAMPLES;
145  // convert the value to resistance
146  reading = (float) 1023 / reading - 1;
147  reading = SERIESRESISTOR / reading;
148 
149  float steinhart;
150  steinhart = reading / THERMISTORNOMINAL; // (R/Ro)
151  steinhart = log(steinhart); // ln(R/Ro)
152  steinhart /= BCOEFFICIENT; // 1/B * ln(R/Ro)
153  steinhart += 1.0 / (TEMPERATURENOMINAL + 273.15); // + (1/To)
154  steinhart = 1.0 / steinhart; // Invert
155  steinhart -= 273.15; // convert to C
156  return steinhart;
157 }

308 void setup()
309 {

311  char conversion[MAX_COLLECTION_SIZE];

325  StaticJsonBuffer<MAX_COLLECTION_SIZE> jsonBuffer;
326  JsonObject& root = jsonBuffer.parseObject(serial_buffer);
327  if (root.success()) {

335  PHPIN = strtol(&ph[1], 0, 10);
336  }
337  if (root.containsKey("daylight") && root.containsKey("standard")) {
338  daylight = root["daylight"];
339  standard = root["standard"];
340  }
341  if (root.containsKey("ip_address")) {
342  strncpy((char*) IP_ADDRESS, root["ip_address"], sizeof(IP_ADDRESS));
343  }

345  }
346  }
347  logToSerial("thermistor pin set to: %d", THERMISTORPIN);
348  logToSerial("ph pin set to: %d", PHPIN);
