17 ## The SerialMonitor manages JSON encoded communications with the alamode
18 #
19 # Currently bi-directional communications are limited to an initial message
20 # to the alamode containing configration details followed by a series of
21 # reads from the alamode containing sensor measurements and logs from the
22 # alamode that terminates on when the fishtank_monitor is stopped. All
23 # communications are encoded in simple JSON strings. Logs entries from the
24 # alamode are prefixed to distinguish them from the fishtank_monitor logs
25 # but are emitted to those logs also. The SerialMonitor runs in its own
26 # thread.

56  ## Given a string read from the alamode, construct a JSON object from it
57  #
58  # The input is parsed and any extraneous characters (prefixes or unexpected
59  # appendices) are discarded. Sensor readings are assumed to be complete -
60  # that is, when one appears, all sensor readings will appear - but log and
61  # sensor entries can appear together in a message. Log entries are
62  # immediately send to the fishtank_monitor log with a prefix making them
63  # easily distinguishable.
64  #
65  # @input the string read from the serial sevice
66  # @return a JSON object representing the message from the alamode
67  def _parse_input(self, input):
68  if '{' in input and '}' in input and (input.index('{') < input.rfind('}')):
69  logger.debug("input is %r" %input)
70  message = json.loads(input[input.index('{') : input.rfind('}') + 1])
71  logger.debug("received from alamode json is %r" %message)
72  if 'temperature' in message and 'ph' in message:
73  logger.info("measured temperature: %r" %message['temperature'])
74  logger.info("measured ph: %r" %message['ph'])
75  if 'log' in message:
76  logger.info("ALAMODE: %s" %message.pop("log"))
77  return message
78  return None

80  ## Endlessly read from serial and make measurements available to third parties
81  #
82  # The run method reads messages from the serial device and sets the sensor
83  # measurements on itself in public members which parties interested in the
84  # most recent observations can read.
