112 // Measure temperature from the thermistor, taking multiple readings to
113 // arrive at a more consistent value
114 float getTemp()
115 {
116  float reading = 0.0;
117  int i = 0;
118  for (i = 0; i<NUM_TEMP_SAMPLES; i++)
119  {
120  float next = analogRead(THERMISTORPIN);
121  reading += next;
122  delay(WAIT_BETWEEN_TEMP_SAMPLES);
123  }
124  reading /= (float) NUM_TEMP_SAMPLES;
125  // convert the value to resistance
126  reading = (float) 1023 / reading - 1;
127  reading = SERIESRESISTOR / reading;
128 
129  float steinhart;
130  steinhart = reading / THERMISTORNOMINAL; // (R/Ro)
131  steinhart = log(steinhart); // ln(R/Ro)
132  steinhart /= BCOEFFICIENT; // 1/B * ln(R/Ro)
133  steinhart += 1.0 / (TEMPERATURENOMINAL + 273.15); // + (1/To)
134  steinhart = 1.0 / steinhart; // Invert
135  steinhart -= 273.15; // convert to C
136  return steinhart;
137 }

139 float getPh()
140 {
141  float ph=0.0;
142  float voltage=0.0;
143  float measurement=0.0;
144  float next = 0.0;
145  int i = 0;
146  for (i=0; i<NUM_PH_SAMPLES; i++)
147  {
148  next = analogRead(PHPIN);
149  measurement += next;
150  delay(WAIT_BETWEEN_PH_SAMPLES);
151  }
152  measurement /= (float) NUM_PH_SAMPLES;
153  voltage = measurement * 5.0 / 1024.0;
154  ph = 3.5 * voltage + OFFSET;
155  return ph;
156 }
