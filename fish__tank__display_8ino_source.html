137 float getTemp()
138 {
139  float reading = 0.0;
140  int i = 0;
141  for (i = 0; i<NUM_TEMP_SAMPLES; i++)
142  {
143  float next = analogRead(THERMISTORPIN);
144  reading += next;
145  delay(WAIT_BETWEEN_TEMP_SAMPLES);
146  }
147  reading /= (float) NUM_TEMP_SAMPLES;
148  // convert the value to resistance
149  reading = (float) 1023 / reading - 1;
150  reading = SERIESRESISTOR / reading;
151 
152  float steinhart;
153  steinhart = reading / THERMISTORNOMINAL; // (R/Ro)
154  steinhart = log(steinhart); // ln(R/Ro)
155  steinhart /= BCOEFFICIENT; // 1/B * ln(R/Ro)
156  steinhart += 1.0 / (TEMPERATURENOMINAL + 273.15); // + (1/To)
157  steinhart = 1.0 / steinhart; // Invert
158  steinhart -= 273.15; // convert to C
159  return steinhart;
160 }

167 float getPh()

178  delay(WAIT_BETWEEN_PH_SAMPLES);
