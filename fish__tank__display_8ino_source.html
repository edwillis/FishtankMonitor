122 float getTemp()
123 {
124  float reading = 0.0;
125  int i = 0;
126  for (i = 0; i<NUM_TEMP_SAMPLES; i++)
127  {
128  float next = analogRead(THERMISTORPIN);
129  reading += next;
130  delay(WAIT_BETWEEN_TEMP_SAMPLES);
131  }
132  reading /= (float) NUM_TEMP_SAMPLES;
133  // convert the value to resistance
134  reading = (float) 1023 / reading - 1;
135  reading = SERIESRESISTOR / reading;
136 
137  float steinhart;
138  steinhart = reading / THERMISTORNOMINAL; // (R/Ro)
139  steinhart = log(steinhart); // ln(R/Ro)
140  steinhart /= BCOEFFICIENT; // 1/B * ln(R/Ro)
141  steinhart += 1.0 / (TEMPERATURENOMINAL + 273.15); // + (1/To)
142  steinhart = 1.0 / steinhart; // Invert
143  steinhart -= 273.15; // convert to C
144  return steinhart;
145 }
