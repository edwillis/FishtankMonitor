138 float getTemp()
139 {
140  float reading = 0.0;
141  int i = 0;
142  for (i = 0; i<NUM_TEMP_SAMPLES; i++)
143  {
144  float next = analogRead(THERMISTORPIN);
145  reading += next;
146  delay(WAIT_BETWEEN_TEMP_SAMPLES);
147  }
148  reading /= (float) NUM_TEMP_SAMPLES;
149  // convert the value to resistance
150  reading = (float) 1023 / reading - 1;
151  reading = SERIESRESISTOR / reading;
152 
153  float steinhart;
154  steinhart = reading / THERMISTORNOMINAL; // (R/Ro)
155  steinhart = log(steinhart); // ln(R/Ro)
156  steinhart /= BCOEFFICIENT; // 1/B * ln(R/Ro)
157  steinhart += 1.0 / (TEMPERATURENOMINAL + 273.15); // + (1/To)
158  steinhart = 1.0 / steinhart; // Invert
159  steinhart -= 273.15; // convert to C
160  return steinhart;
161 }

168 float getPh()
169 {
170  float ph=0.0;
171  float voltage=0.0;
172  float measurement=0.0;
173  float next = 0.0;
174  int i = 0;

176  {
177  next = analogRead(PHPIN);
178  measurement += next;
179  delay(WAIT_BETWEEN_PH_SAMPLES);

182  voltage = measurement * 5.0 / 1024.0;
183  ph = 3.5 * voltage + (float) OFFSET;

262  daylight = root["daylight"];

267  logToSerial("thermistor pin set to: %d", THERMISTORPIN);
268  logToSerial("ph pin set to: %d", PHPIN);

270  TimeChangeRule DT = {"DT", Second, Sun, Mar, 2, daylight};

281 void display(int showSerial)

289  tempCelLeft += 1;
290  }

292  int tempFarLeft = tempF / 10;
293  int tempFarRight = tempF % 10;
294  if (tempFarRight >= 5)
295  {
296  tempFarLeft += 1;
297  }
298  printPseudoFloat(lcd, tempFarLeft, 0, "F", 0);

315  }
