17 ## The SerialMonitor manages JSON encoded communications with the alamode
18 #
19 # The communications protocol between the Alamode and the Pi starts with the
20 # Pi sending the Alamode configuration data (IP address to display, sensor
21 # configuration and offsets etc) to the Alamode after which communications
22 # are symmetric, with the Alamode sending data of some form (e.g. sensor
23 # readings, log entries etc) and the Pi immediately responding with the
24 # (possibly updated) configuration data. This cycle only terminatesc when
25 # the fishtank_monitor is stopped. All communications are encoded in simple
26 # JSON strings. Logs entries from the alamode are prefixed to distinguish
27 # them from the fishtank_monitor logs but are emitted to those logs also.
28 # The SerialMonitor runs in its own thread.

67  ## Given a string read from the Alamode, construct a JSON object from it
68  #
69  # The input is parsed and any extraneous characters (prefixes or unexpected
70  # appendices) are discarded. Sensor readings are assumed to be complete -
71  # that is, when one appears, all sensor readings will appear - but log and
72  # sensor entries can appear together in a message. Log entries are
73  # immediately send to the fishtank_monitor log with a prefix making them
74  # easily distinguishable.
75  #
76  # @input the string read from the serial sevice
77  # @return a JSON object representing the message from the alamode
78  def _parse_input(self, input):
79  if '{' in input and '}' in input and (input.index('{') < input.rfind('}')):
80  logger.debug("input is %r" %input)
81  message = json.loads(input[input.index('{') : input.rfind('}') + 1])
82  logger.debug("received from alamode json is %r" %message)
83  if 'temperature' in message and 'ph' in message:
84  logger.info("measured temperature: %r" %message['temperature'])
85  logger.info("measured ph: %r" %message['ph'])
86  if 'log' in message:
87  logger.info("ALAMODE: %s" %message.pop("log"))
88  return message
89  return None

91  ## Manage the Pi to Alamode communications protocol
92  #
93  # The run method reads messages from the serial device and sets the sensor
94  # measurements on itself in public members which parties interested in the
95  # most recent observations can read. Each received message trigger the send
96  # of configuration data back to the Alamode.
97  def run(self):
98  try:
99  stop = False
100  while not stop:
101  next = self.ard.readline().decode('UTF8')
102  if next:
103  logger.debug("serial raw read %r"%next)
104  message = self._parse_input(next)
105  if message and 'temperature' in message and 'ph' in message:
106  self.temperature = message['temperature']
107  self.ph = message['ph']
108  if not self.started.is_set():
109  self.started.set()
110  self._write_to_serial(self.configuration)

112  except Exception as e:
