62 #
63 # Examine the current ph and temperature values and warn the user by email if
64 # they exceed limits
65 class NotifyWarnings(NotifierBase):

67  ## The last time we warned the user by email
68  time_last_warned = 0
69 
70  def __call__(self, conn, monitor):
71  ph_bad = False
72  temp_bad = False
73  msg = ''
74  if monitor.ph and (monitor.ph < 6.0 or monitor.ph > 8.0):
75  logger.warning("ph is bad: %r" %monitor.ph)
76  ph_bad = True
77  if monitor.temperature and (monitor.temperature < 20 or monitor.temperature > 28):
78  logger.warning("temperature is bad: %r" %monitor.temperature)
79  temp_bad = True
80  if ph_bad:
81  msg += 'Fishtank PH level is unsafe: %r\n'%monitor.ph
82  logger.warn("unsafe ph, will email")
83  if temp_bad:
84  msg += 'Fishtank temperature is unsafe: %r\n'%monitor.temperature
85  logger.warn("unsafe temperature, will email")
86  if msg and time.time() - self.time_last_warned > config.send_warnings_interval:
87  self.time_last_warned = time.time()
88  logger.info("setting time_last_warned to %r" %self.time_last_warned)
89  msg += "\nWarning emailed notifications will continue every %r hours until the readings fall within expected ranges."\

91  if config.send_warnings_interval > 0:
