140 float getTemp()
141 {
142  float reading = 0.0;
143  int i = 0;
144  for (i = 0; i<NUM_TEMP_SAMPLES; i++)
145  {
146  float next = analogRead(THERMISTORPIN);
147  reading += next;
148  delay(WAIT_BETWEEN_TEMP_SAMPLES);
149  }
150  reading /= (float) NUM_TEMP_SAMPLES;
151  // convert the value to resistance
152  reading = (float) 1023 / reading - 1;
153  reading = SERIESRESISTOR / reading;
154 
155  float steinhart;
156  steinhart = reading / THERMISTORNOMINAL; // (R/Ro)
157  steinhart = log(steinhart); // ln(R/Ro)
158  steinhart /= BCOEFFICIENT; // 1/B * ln(R/Ro)
159  steinhart += 1.0 / (TEMPERATURENOMINAL + 273.15); // + (1/To)
160  steinhart = 1.0 / steinhart; // Invert
161  steinhart -= 273.15; // convert to C
162  return steinhart;
163 }

326 void receiveConfigurationOverSerial() {

328  int receivedConfiguration = 0;
329  while (!receivedConfiguration)

334  JsonObject& root = jsonBuffer.parseObject(serial_buffer);
335  if (root.success()) {

340  const char* temp = root["thermistor_pin"];
341  const char* ph = root["ph_pin"];
342  THERMISTORPIN = strtol(&temp[1], 0, 10);
343  PHPIN = strtol(&ph[1], 0, 10);
344  }
345  if (root.containsKey("daylight") && root.containsKey("standard")) {
