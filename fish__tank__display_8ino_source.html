133 float getTemp()
134 {
135  float reading = 0.0;
136  int i = 0;
137  for (i = 0; i<NUM_TEMP_SAMPLES; i++)
138  {
139  float next = analogRead(THERMISTORPIN);
140  reading += next;
141  delay(WAIT_BETWEEN_TEMP_SAMPLES);
142  }
143  reading /= (float) NUM_TEMP_SAMPLES;
144  // convert the value to resistance
145  reading = (float) 1023 / reading - 1;
146  reading = SERIESRESISTOR / reading;
147 
148  float steinhart;
149  steinhart = reading / THERMISTORNOMINAL; // (R/Ro)
150  steinhart = log(steinhart); // ln(R/Ro)
151  steinhart /= BCOEFFICIENT; // 1/B * ln(R/Ro)
152  steinhart += 1.0 / (TEMPERATURENOMINAL + 273.15); // + (1/To)
153  steinhart = 1.0 / steinhart; // Invert
154  steinhart -= 273.15; // convert to C
155  return steinhart;
156 }

163 float getPh()

165  float ph=0.0;
166  float voltage=0.0;
167  float measurement=0.0;

169  int i = 0;

171  {
172  next = analogRead(PHPIN);
173  measurement += next;
174  delay(WAIT_BETWEEN_PH_SAMPLES);

176  measurement /= (float) NUM_PH_SAMPLES;
